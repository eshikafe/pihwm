141  /* GPIO18= PWM channel-A Funct. 5

146  /* Derive PWM clock direct from X-tal

148  The values below depends on the X-tal frequency! */
149  PWMCLK_DIV = 0x5A000000 | (32 << 12);/* set pwm div to 32 (19.2/3 = 600KHz) */
